    function change(total, coin) {

        var arr = [1];
    
        for (var i = 1; i <= total; i++) arr[i] = 0;
    
        for (var i = 0; i < coin.length; i++) {
            for (var j = 1; j <= total; j++)
                if (coin[i] <= j)
                    arr[j] += arr[j - coin[i]];
        }
    
        return arr[total];
    }
    
    // 아래는 테스트로 출력해 보기 위한 코드입니다.
    console.log(change(1000, [48, 23,3]));
